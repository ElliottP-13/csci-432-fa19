\section{Asymptotics}

\begin{enumerate}
\item Show $f(n)=n$ is $O(\frac{1}{2}n)$. \vspace{1in}
\item Show $f(n)=3n$ is $O(n^3)$. \vspace{1in}
\item Show $f(n)=n^2+n$ is $O(n^2)$. \vspace{1in}
\item Show $f(n)=n\log n +1$ is $O(n^2)$. \vspace{1in}
\end{enumerate}
